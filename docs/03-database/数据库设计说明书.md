## 1. 数据库概述

### 1.1 设计原则
- **标准化**: 遵循第三范式（3NF），减少数据冗余
- **扩展性**: 预留扩展字段，支持未来业务变化
- **安全性**: 敏感数据加密存储，审计字段完整
- **性能**: 合理设计索引，支持大数据量查询

## 2. 系统管理模块

### 2.1 用户表 (users)

```sql
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username VARCHAR(50) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    real_name VARCHAR(50) NOT NULL,
    email VARCHAR(100),
    phone VARCHAR(20),
    avatar_url VARCHAR(255),
    department_id UUID REFERENCES departments(id),
    status SMALLINT NOT NULL DEFAULT 1, -- 0:禁用, 1:启用
    last_login_at TIMESTAMP,
    last_login_ip VARCHAR(50),
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created_by UUID REFERENCES users(id),
    updated_by UUID REFERENCES users(id)
);

CREATE INDEX idx_users_dept ON users(department_id);
CREATE INDEX idx_users_status ON users(status);
```

### 2.2 角色表 (roles)

```sql
CREATE TABLE roles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    role_code VARCHAR(50) NOT NULL UNIQUE,
    role_name VARCHAR(50) NOT NULL,
    description VARCHAR(255),
    data_scope SMALLINT DEFAULT 1, -- 1:全部, 2:本部门, 3:本人
    status SMALLINT NOT NULL DEFAULT 1,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);
```

### 2.3 权限表 (permissions)

```sql
CREATE TABLE permissions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    perm_code VARCHAR(100) NOT NULL UNIQUE,
    perm_name VARCHAR(100) NOT NULL,
    parent_id UUID REFERENCES permissions(id),
    perm_type SMALLINT NOT NULL, -- 1:菜单, 2:按钮, 3:接口
    path VARCHAR(255),
    component VARCHAR(255),
    icon VARCHAR(50),
    sort_order INTEGER DEFAULT 0,
    status SMALLINT NOT NULL DEFAULT 1
);

CREATE INDEX idx_permissions_parent ON permissions(parent_id);
```

### 2.4 用户角色关联表 (user_roles)

```sql
CREATE TABLE user_roles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    role_id UUID NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(user_id, role_id)
);
```

### 2.5 角色权限关联表 (role_permissions)

```sql
CREATE TABLE role_permissions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    role_id UUID NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
    permission_id UUID NOT NULL REFERENCES permissions(id) ON DELETE CASCADE,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(role_id, permission_id)
);
```

### 2.6 部门表 (departments)

```sql
CREATE TABLE departments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    dept_code VARCHAR(50) NOT NULL UNIQUE,
    dept_name VARCHAR(50) NOT NULL,
    parent_id UUID REFERENCES departments(id),
    leader_id UUID REFERENCES users(id),
    sort_order INTEGER DEFAULT 0,
    status SMALLINT NOT NULL DEFAULT 1,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);
```

## 3. 头寸管理模块

### 3.1 现货头寸表 (spot_positions)

```sql
CREATE TABLE spot_positions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    position_no VARCHAR(50) NOT NULL UNIQUE,
    position_type SMALLINT NOT NULL, -- 1:库存, 2:销售订单, 3:采购订单
    product_type VARCHAR(20) NOT NULL DEFAULT 'CU', -- 铜
    quantity DECIMAL(18,4) NOT NULL, -- 数量(吨)
    unit_cost DECIMAL(18,4), -- 单位成本
    total_cost DECIMAL(18,4), -- 总成本
    currency VARCHAR(3) DEFAULT 'CNY',
    
    -- 库存特有
    warehouse VARCHAR(100),
    quality_grade VARCHAR(20),
    
    -- 订单特有
    counterparty VARCHAR(100), -- 交易对手
    contract_no VARCHAR(50), -- 合同编号
    delivery_date DATE, -- 交货日期
    pricing_type SMALLINT, -- 定价方式: 1:固定价, 2:点价, 3:均价
    base_price DECIMAL(18,4), -- 基准价格
    premium DECIMAL(18,4), -- 升贴水
    final_price DECIMAL(18,4), -- 最终价格
    
    -- 点价状态
    pricing_status SMALLINT DEFAULT 0, -- 0:未点价, 1:部分点价, 2:已点价
    priced_quantity DECIMAL(18,4) DEFAULT 0, -- 已点价数量
    
    -- 状态
    status SMALLINT NOT NULL DEFAULT 1, -- 0:已关闭, 1:有效
    hedge_status SMALLINT DEFAULT 0, -- 0:未保值, 1:部分保值, 2:已保值
    hedged_quantity DECIMAL(18,4) DEFAULT 0, -- 已保值数量
    
    -- 时间戳
    position_date DATE NOT NULL, -- 头寸日期
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created_by UUID REFERENCES users(id),
    updated_by UUID REFERENCES users(id),
    
    -- 备注
    remark TEXT
);

CREATE INDEX idx_spot_status ON spot_positions(status);
CREATE INDEX idx_spot_type ON spot_positions(position_type);
CREATE INDEX idx_spot_date ON spot_positions(position_date);
CREATE INDEX idx_spot_contract ON spot_positions(contract_no);
```

### 3.2 库存明细表 (inventory)

```sql
CREATE TABLE inventory (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    position_id UUID REFERENCES spot_positions(id),
    warehouse VARCHAR(100) NOT NULL,
    product_type VARCHAR(20) NOT NULL,
    quantity DECIMAL(18,4) NOT NULL,
    unit_cost DECIMAL(18,4),
    quality_grade VARCHAR(20),
    batch_no VARCHAR(50),
    entry_date DATE,
    status SMALLINT DEFAULT 1,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_inventory_warehouse ON inventory(warehouse);
```

### 3.3 点价记录表 (pricing_records)

```sql
CREATE TABLE pricing_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id UUID NOT NULL REFERENCES spot_positions(id),
    pricing_date DATE NOT NULL,
    priced_quantity DECIMAL(18,4) NOT NULL,
    base_price DECIMAL(18,4) NOT NULL, -- 点价时的期货价格
    premium DECIMAL(18,4) NOT NULL, -- 升贴水
    final_price DECIMAL(18,4) NOT NULL, -- 最终结算价
    futures_contract VARCHAR(20), -- 对应的期货合约
    executed_by UUID REFERENCES users(id),
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    remark TEXT
);
```

### 3.4 期货持仓表 (futures_positions)

```sql
CREATE TABLE futures_positions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    position_no VARCHAR(50) NOT NULL UNIQUE,
    account_id UUID NOT NULL REFERENCES accounts(id),
    
    -- 合约信息
    exchange VARCHAR(20) NOT NULL, -- 交易所: SHFE
    contract_code VARCHAR(20) NOT NULL, -- 合约代码: CU2503
    product_type VARCHAR(20) NOT NULL DEFAULT 'CU',
    
    -- 持仓方向
    direction SMALLINT NOT NULL, -- 1:买(多), 2:卖(空)
    
    -- 持仓数量
    total_quantity DECIMAL(18,4) NOT NULL, -- 总持仓
    available_quantity DECIMAL(18,4) NOT NULL, -- 可用持仓
    frozen_quantity DECIMAL(18,4) DEFAULT 0, -- 冻结持仓
    
    -- 价格
    open_price DECIMAL(18,4) NOT NULL, -- 开仓均价
    settlement_price DECIMAL(18,4), -- 结算价
    last_price DECIMAL(18,4), -- 最新价
    
    -- 盈亏
    position_profit DECIMAL(18,4) DEFAULT 0, -- 持仓盈亏
    close_profit DECIMAL(18,4) DEFAULT 0, -- 平仓盈亏
    total_profit DECIMAL(18,4) DEFAULT 0, -- 总盈亏
    
    -- 保证金
    margin_occupied DECIMAL(18,4) DEFAULT 0, -- 占用保证金
    margin_ratio DECIMAL(8,4), -- 保证金比例
    
    -- 套保属性
    hedge_type SMALLINT DEFAULT 1, -- 1:套保, 2:投机
    scheme_id UUID REFERENCES hedge_schemes(id), -- 关联套保方案
    
    -- 状态
    status SMALLINT NOT NULL DEFAULT 1, -- 0:已平仓, 1:持仓中
    open_date DATE NOT NULL, -- 开仓日期
    expire_date DATE, -- 到期日期
    
    -- 交割信息
    delivery_intent BOOLEAN DEFAULT FALSE, -- 交割意向
    delivery_status SMALLINT DEFAULT 0, -- 0:无, 1:意向, 2:配对, 3:完成
    
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created_by UUID REFERENCES users(id),
    updated_by UUID REFERENCES users(id),
    
    remark TEXT
);

CREATE INDEX idx_futures_account ON futures_positions(account_id);
CREATE INDEX idx_futures_contract ON futures_positions(contract_code);
CREATE INDEX idx_futures_status ON futures_positions(status);
CREATE INDEX idx_futures_scheme ON futures_positions(scheme_id);
```

### 3.5 期货账户表 (accounts)

```sql
CREATE TABLE accounts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    broker_name VARCHAR(50) NOT NULL, -- 期货公司
    exchange VARCHAR(20) NOT NULL, -- 交易所

    balance DECIMAL(18,4) DEFAULT 0, -- 权益

    margin_occupied DECIMAL(18,4) DEFAULT 0, -- 占用保证金
    frozen_margin DECIMAL(18,4) DEFAULT 0, -- 冻结保证金
    risk_ratio DECIMAL(8,4) DEFAULT 0, -- 风险度
    
    -- 额度
    total_quota DECIMAL(18,4), -- 总套保额度
    used_quota DECIMAL(18,4) DEFAULT 0, -- 已用额度
```

## 4. 套保方案模块

### 4.1 套保方案表 (hedge_schemes)

```sql
CREATE TABLE hedge_schemes (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    scheme_no VARCHAR(50) NOT NULL UNIQUE, -- 方案编号
    scheme_name VARCHAR(100) NOT NULL, -- 方案名称
    
    -- 套保标的
    product_type VARCHAR(20) NOT NULL DEFAULT 'CU',
    spot_exposure DECIMAL(18,4) NOT NULL, -- 现货敞口数量
    
    -- 套保策略
    hedge_ratio DECIMAL(5,2) NOT NULL, -- 套保比例(%)
    target_quantity DECIMAL(18,4) NOT NULL, -- 目标保值数量
    executed_quantity DECIMAL(18,4) DEFAULT 0, -- 已执行数量
    
    -- 合约选择
    target_contract VARCHAR(20), -- 目标合约
    strategy_type SMALLINT DEFAULT 1, -- 1:直接套保, 2:展期套保
    
    -- 建仓策略
    build_strategy SMALLINT DEFAULT 1, -- 1:一次性, 2:分批建仓
    price_range_low DECIMAL(18,4), -- 目标价格区间低
    price_range_high DECIMAL(18,4), -- 目标价格区间高
    
    -- 风险控制
    stop_loss_price DECIMAL(18,4), -- 止损价格
    take_profit_price DECIMAL(18,4), -- 止盈价格
    
    -- 时间
    start_date DATE NOT NULL, -- 开始日期
    end_date DATE NOT NULL, -- 结束日期
    
    -- 审批流程
    status SMALLINT NOT NULL DEFAULT 0, -- 0:草稿, 1:审批中, 2:已通过, 3:已拒绝, 4:执行中, 5:已完成, 6:已终止
    current_approver UUID REFERENCES users(id), -- 当前审批人
    approved_at TIMESTAMP, -- 审批通过时间
    
    -- 执行结果
    actual_quantity DECIMAL(18,4) DEFAULT 0, -- 实际保值数量
    avg_price DECIMAL(18,4), -- 建仓均价
    close_quantity DECIMAL(18,4) DEFAULT 0, -- 已平仓数量
    close_avg_price DECIMAL(18,4), -- 平仓均价
    
    -- 效果评估
    hedge_effectiveness DECIMAL(8,4), -- 套保有效性
    total_pnl DECIMAL(18,4), -- 总盈亏
    
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created_by UUID REFERENCES users(id),
    updated_by UUID REFERENCES users(id),
    
    remark TEXT
);

CREATE INDEX idx_scheme_status ON hedge_schemes(status);
CREATE INDEX idx_scheme_date ON hedge_schemes(start_date, end_date);
CREATE INDEX idx_scheme_creator ON hedge_schemes(created_by);
```

## 5. 交易执行模块

### 5.1 交易指令表 (trade_orders)

```sql
CREATE TABLE trade_orders (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_no VARCHAR(50) NOT NULL UNIQUE, -- 指令编号
    
    -- 指令类型
    order_type SMALLINT NOT NULL, -- 1:开仓, 2:平仓, 3:移仓
    direction SMALLINT NOT NULL, -- 1:买入, 2:卖出
    
    -- 合约信息
    exchange VARCHAR(20) NOT NULL,
    contract_code VARCHAR(20) NOT NULL,
    product_type VARCHAR(20) NOT NULL DEFAULT 'CU',
    
    -- 数量价格
    quantity DECIMAL(18,4) NOT NULL, -- 指令数量
    order_price_type SMALLINT DEFAULT 1, -- 1:限价, 2:市价, 3:条件单
    limit_price DECIMAL(18,4), -- 限价
    
    -- 关联信息
    scheme_id UUID REFERENCES hedge_schemes(id), -- 关联套保方案
    account_id UUID REFERENCES accounts(id), -- 交易账户
    position_id UUID REFERENCES futures_positions(id), -- 关联持仓(平仓时)
    
    -- 指令状态
    status SMALLINT NOT NULL DEFAULT 0, -- 0:草稿, 1:待审批, 2:审批通过, 3:审批拒绝, 4:执行中, 5:已完成, 6:已撤销
    
    -- 审批
    require_approval BOOLEAN DEFAULT TRUE,
    approved_by UUID REFERENCES users(id),
    approved_at TIMESTAMP,
    
    -- 执行
    executed_quantity DECIMAL(18,4) DEFAULT 0, -- 已成交数量
    avg_price DECIMAL(18,4), -- 成交均价
    executed_at TIMESTAMP, -- 完成时间
    executed_by UUID REFERENCES users(id), -- 执行交易员
    
    -- 时间
    valid_until TIMESTAMP, -- 有效截止时间
    
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created_by UUID REFERENCES users(id),
    updated_by UUID REFERENCES users(id),
    
    remark TEXT
);

CREATE INDEX idx_order_status ON trade_orders(status);
CREATE INDEX idx_order_scheme ON trade_orders(scheme_id);
CREATE INDEX idx_order_account ON trade_orders(account_id);
```

```sql
CREATE TABLE risk_warnings (

    -- 关联对象

    threshold_value DECIMAL(18,4), -- 阈值
    current_value DECIMAL(18,4), -- 当前值
    
    -- 处理状态
    status SMALLINT DEFAULT 0, -- 0:未处理, 1:处理中, 2:已处理, 3:已忽略
    handled_by UUID REFERENCES users(id),
    handled_at TIMESTAMP,
    handle_result TEXT,
    
    -- 通知
    notified BOOLEAN DEFAULT FALSE,
    
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP

CREATE INDEX idx_warning_status ON risk_warnings(status);

```

### 6.2 风控规则表 (risk_rules)

```sql
CREATE TABLE risk_rules (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    rule_code VARCHAR(50) NOT NULL UNIQUE,
    rule_name VARCHAR(100) NOT NULL,
    rule_type VARCHAR(50) NOT NULL, -- MARGIN, RISK, POSITION, QUOTA
    
    -- 规则条件
    condition_field VARCHAR(50) NOT NULL, -- 字段
    condition_operator VARCHAR(20) NOT NULL, -- 操作符: >, <, =, >=, <=
    threshold_value DECIMAL(18,4) NOT NULL, -- 阈值
    
    -- 预警级别
    warning_level SMALLINT NOT NULL, -- 1:提醒, 2:预警, 3:告警
```
